#include <stdio.h>
#define maxsize 100

typedef struct stack {
    int items[maxsize];
    int top;
} stack;

void initialize(stack *stack) {
    stack->top = -1;
}

int is_empty(stack *stack) {
    return (stack->top == -1);
}

int is_full(stack *stack) {
    return (stack->top == maxsize - 1);
}

void push(stack *stack, int val) {
    if (is_full(stack)) {
        printf("Stack overflow\n");
    } else {
        stack->top++;
        stack->items[stack->top] = val;
    }
}

int pop(stack *stack) {
    if (is_empty(stack)) {
        printf("Stack is empty\n");
        return -1;
    } else {
        int val = stack->items[stack->top];
        stack->top--;
        return val;
    }
}

int main() {
    stack stack1;
    initialize(&stack1);
    int i = 1;
    int j = 1;

    while (i < maxsize) {
        int value;
        printf("Push value: ");
        scanf("%d", &value);
        push(&stack1, value);
        i++;
    }

    printf("The stack is:\n");
    while (j < maxsize) {
        printf("Popped value: %d\n", pop(&stack1));
        j++;
    }

    return 0;
}

// Time Complexity: Push O(1), Pop O(1)
// Space Complexity: O(maxsize)
