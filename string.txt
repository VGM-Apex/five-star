#include <stdio.h>

// Function to calculate string length
int stringLength(char str[]) {
    int len = 0;
    while (str[len] != '\0') len++;
    return len;
}

// Function to compare two strings
int stringCompare(char str1[], char str2[]) {
    int i = 0;
    while (str1[i] != '\0' && str2[i] != '\0') {
        if (str1[i] != str2[i])
            return str1[i] - str2[i];
        i++;
    }
    return str1[i] - str2[i];
}

// Function to copy a string
void stringCopy(char dest[], char src[]) {
    int i = 0;
    while (src[i] != '\0') {
        dest[i] = src[i];
        i++;
    }
    dest[i] = '\0';
}

// Function to reverse a string
void stringReverse(char str[]) {
    int len = stringLength(str);
    for (int i = 0; i < len / 2; i++) {
        char temp = str[i];
        str[i] = str[len - i - 1];
        str[len - i - 1] = temp;
    }
}

int main() {
    char str1[100], str2[100], str3[100];

    printf("Enter first string: ");
    scanf("%s", str1);
    printf("Enter second string: ");
    scanf("%s", str2);

    // String Comparison
    int cmp = stringCompare(str1, str2);
    if (cmp == 0)
        printf("Strings are equal\n");
    else if (cmp < 0)
        printf("First string is less than second string\n");
    else
        printf("First string is greater than second string\n");

    // String Copy
    stringCopy(str3, str1);
    printf("Copied string: %s\n", str3);

    // String Length
    printf("Length of first string: %d\n", stringLength(str1));
    printf("Length of second string: %d\n", stringLength(str2));

    // String Reverse
    stringReverse(str1);
    printf("Reversed first string: %s\n", str1);

    return 0;
}

/*
Time Complexity: Comparison O(n), Copy O(n), Length O(n), Reverse O(n)
Space Complexity: O(n) for copied string, O(1) extra for other operations
*/
