#include <stdio.h>
#include <stdlib.h>

void traversing(int arr[], int size)
{
    printf("The array is:\n");
    for (int i = 0; i < size; i++)
    {
        printf("%d, ", arr[i]);
    }
    printf("\n");
}

int searching(int arr[], int size, int value)
{
    for (int i = 0; i < size; i++)
    {
        if (arr[i] == value)
        {
            return 1;
        }
    }
    return 0;
}

void sorting(int arr[], int size)
{
    for (int i = 0; i < size - 1; i++)
    {
        for (int j = i + 1; j < size; j++)
        {
            if (arr[i] > arr[j])
            {
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

void insertion(int arr[], int *size, int position, int value)
{
    for (int i = *size; i > position; i--)
    {
        arr[i] = arr[i - 1];
    }
    arr[position] = value;
    (*size)++;
}

void deletion(int arr[], int *size, int position)
{
    for (int i = position; i < *size - 1; i++)
    {
        arr[i] = arr[i + 1];
    }
    (*size)--;
}

int main()
{
    int n, position, value;
    printf("Enter the size of array: ");
    scanf("%d", &n);

    int arr[100]; // assuming maximum size 100
    int ch;

    printf("1. For traversing\n");
    printf("2. For searching an element\n");
    printf("3. For sorting the array\n");
    printf("4. For inserting element in array\n");
    printf("5. For deleting an element\n");
    printf("6. Exit\n");

    for (int i = 0; i < n; i++)
    {
        printf("Enter element %d of array: ", i);
        scanf("%d", &arr[i]);
    }

    int size = n;

    while (1)
    {
        printf("Enter operation: ");
        scanf("%d", &ch);
        switch (ch)
        {
        case 1:
            traversing(arr, size);
            break;

        case 2:
            printf("Searching for: ");
            scanf("%d", &value);
            if (searching(arr, size, value))
                printf("Value %d found in the array.\n", value);
            else
                printf("Value %d not found in the array.\n", value);
            break;

        case 3:
            sorting(arr, size);
            printf("Array sorted.\n");
            break;

        case 4:
            printf("At what position you want to enter new element: ");
            scanf("%d", &position);
            printf("Enter value: ");
            scanf("%d", &value);
            insertion(arr, &size, position, value);
            break;

        case 5:
            printf("Which position you want to delete: ");
            scanf("%d", &position);
            deletion(arr, &size, position);
            break;

        case 6:
            return 0;

        default:
            printf("Enter a valid choice.\n");
            break;
        }
    }

    return 0;
}

// Time Complexity: O(n^2) for sorting, O(n) for traversing, searching, insertion, deletion
// Space Complexity: O(1)
