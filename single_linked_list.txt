#include <stdio.h>
#include <stdlib.h>

typedef struct node
{
    int data;
    struct node *next;
} node;

void insertion_beginning(node **head, int value)
{
    node *newNode = (node *)malloc(sizeof(node));
    newNode->data = value;
    newNode->next = *head;
    *head = newNode;
}

void insertion_end(node **head, int value)
{
    node *newNode = (node *)malloc(sizeof(node));
    newNode->data = value;
    newNode->next = NULL;
    if (*head == NULL)
    {
        *head = newNode;
        return;
    }
    node *temp = *head;
    while (temp->next != NULL)
        temp = temp->next;
    temp->next = newNode;
}

void insertion_position(node **head, int value, char c, int number)
{
    node *newNode = (node *)malloc(sizeof(node));
    newNode->data = value;
    newNode->next = NULL;

    if (*head == NULL)
    {
        *head = newNode;
        return;
    }

    node *temp = *head;
    if (c == 'a') // after number
    {
        while (temp != NULL && temp->data != number)
            temp = temp->next;
        if (temp == NULL)
            return;
        newNode->next = temp->next;
        temp->next = newNode;
    }
    else if (c == 'b') // before number
    {
        if ((*head)->data == number)
        {
            newNode->next = *head;
            *head = newNode;
            return;
        }
        node *prev = NULL;
        temp = *head;
        while (temp != NULL && temp->data != number)
        {
            prev = temp;
            temp = temp->next;
        }
        if (temp == NULL)
            return;
        prev->next = newNode;
        newNode->next = temp;
    }
}

void delete_element(node **head, int position)
{
    if (*head == NULL)
        return;

    node *temp = *head;

    if (position == 1)
    {
        *head = temp->next;
        free(temp);
        return;
    }

    for (int i = 1; temp != NULL && i < position - 1; i++)
        temp = temp->next;

    if (temp == NULL || temp->next == NULL)
        return;

    node *next = temp->next->next;
    free(temp->next);
    temp->next = next;
}

int search(node *head, int value)
{
    node *temp = head;
    while (temp != NULL)
    {
        if (temp->data == value)
            return 1;
        temp = temp->next;
    }
    return 0;
}

void display(node *head)
{
    node *temp = head;
    printf("Linked List: ");
    while (temp != NULL)
    {
        printf("%d -> ", temp->data);
        temp = temp->next;
    }
    printf("NULL\n");
}

void freeLinkedList(node *head)
{
    node *temp;
    while (head != NULL)
    {
        temp = head;
        head = head->next;
        free(temp);
    }
}

int main()
{
    node *head = NULL;
    int size, value, val1, choice, numb, position;
    char c;

    while (1)
    {
        printf("0. Create a linked list\n");
        printf("1. Display linked list\n");
        printf("2. Insert at beginning\n");
        printf("3. Insert at end\n");
        printf("4. Insert at position\n");
        printf("5. Delete a node\n");
        printf("6. Search an element\n");
        printf("7. Exit\n");
        printf("Please enter a number: ");
        scanf("%d", &choice);

        switch (choice)
        {
        case 0:
            printf("Enter the number of elements: ");
            scanf("%d", &size);
            for (int i = 0; i < size; i++)
            {
                printf("Value at node %d: ", i + 1);
                scanf("%d", &value);
                insertion_beginning(&head, value);
            }
            break;
        case 1:
            display(head);
            break;
        case 2:
            printf("Enter the value: ");
            scanf("%d", &val1);
            insertion_beginning(&head, val1);
            break;
        case 3:
            printf("Enter the value: ");
            scanf("%d", &val1);
            insertion_end(&head, val1);
            break;
        case 4:
            printf("a: insert after the number\nb: insert before the number\n");
            scanf(" %c", &c);
            printf("Which number: ");
            scanf("%d", &numb);
            printf("Enter the value: ");
            scanf("%d", &val1);
            insertion_position(&head, val1, c, numb);
            break;
        case 5:
            printf("Which position do you want to delete: ");
            scanf("%d", &position);
            delete_element(&head, position);
            break;
        case 6:
            printf("Searching for: ");
            scanf("%d", &val1);
            if (search(head, val1))
                printf("Element is in the list\n");
            else
                printf("Element is not in the list\n");
            break;
        case 7:
            freeLinkedList(head);
            return 0;
        default:
            printf("Enter a valid number\n");
            break;
        }
    }
}

// Time Complexity: Insertion, Deletion, Search O(n), Display O(n)
// Space Complexity: O(1)
